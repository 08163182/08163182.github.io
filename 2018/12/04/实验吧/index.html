import requests
import base64
url="http://ctf5.shiyanbar.com/web/10/10.php"
r=requests.get(url)
if 'FLAG'in r.headers:
        x=bytes(r.headers['FLAG'],"utf-8")
        key=base64.b64decode(x).decode('utf-8').split(':')[1]
        data={'key':key}
        print(requests.post(url,data=data).text)
