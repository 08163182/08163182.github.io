#Web客户端和服务器

##urlparse
处理URL字符串： 

import urllib.parse
print(urllib.parse.urlparse('http://www.python.org/doc/FAQ.html'))

ParseResult(scheme='http', netloc='www.python.org', path='/doc/FAQ.html', params='', query='')
scheme: 下载方案
path:/分割文件或CGI应用路径
params:可选参数
query：连接符&分割的键值对
frag：指定文档特定锚部门

net_loc字符串： user:passwd@host:port
host:远程Web服务器计算机名称或地址 port:端口号

##urljoin
import urllib.parse
urllib.parse.urljoin(‘http://www.python.org/doc/FAQ.html','current/lib/lib.htm')
net_loc及其前面完整路径与newurl链接

##urlopen()
urlopen(urlstr, postQueryData=None)
打开一个给定URL字符串表示的Web连接，连接成功，返回一个文件类型对象，文件对象为f，可获得方法f.read(),f.readline(),f.readlines(),f.close(),f.fileno()#返回文件句柄,f.geturl()#返回真正url，

import urllib.request
urllib.request.urlopen( )

##urlretrieve
urlretrieve(url, filename=None, reporthook=None, data=None)

##quote(),quote_plus()
quote()获取URL数据并编码，用于URL字符串
quote(urldata, safe=’/‘)
quote_plus()可将空格编码成’+’

##unquote(),unquote_plus()
unquote(urldata)
URL字母解码,返回字符串

##urlencode()
urlencode将接受的字典的键值对进行编译

HTTP验证脚本

import urllib.request, urllib.error, urllib.parse

LOGIN ='wesley'
PASSWD = "you'llNeverGuess"
URL = 'http://localhost'
REALM = 'Secure Archive'

def handler_version(url):
       hdlr = urllib.request.HTTPBasicAuthHandler()
    hdlr.add_password(REALM,urllib.parse.urlparse(url)[1], LOGIN, PASSWD)
    opener = urllib.request.build_opener(hdlr)
    urllib.request.install_opener(opener)
    return url

def request_version(url):
    from base64 import encodestring
    req = urllib.request.Request(url)
    b64str= encodestring(bytes ('%s : %s' % (LOGIN, PASSWD), 'utf-8'))[:-1]
    req.add_header("Authorization", "Basic %s" % b64str)
    return req

for funcType in('handler', 'request'):
    print('*** Using %s:' % funcType.upper())
    url = eval('%s_version' % funcType)(URL)
    f = urllib.request.urlopen(url)
    print(str(f.readline(), 'utf-8'))
    f.close()
